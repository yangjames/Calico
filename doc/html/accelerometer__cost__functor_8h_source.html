28 // Generic auto-differentiation accelerometer cost functor. Residuals will be based on
29 // how the accelerometer model is initialized.
30 class AccelerometerCostFunctor {
31  public:
32  static constexpr int kAccelerometerResidualSize = 3;
33  explicit AccelerometerCostFunctor(
34  AccelerometerIntrinsicsModel accelerometer_model,

43  Eigen::Vector3d& gravity, Trajectory& trajectory_world_sensorrig,
44  double stamp, std::vector<double*>& parameters);

46  // Parameters to the cost function:
47  // intrinsics:
48  // All parameters in the intrinsics model as an Eigen column vector.
49  // Order of the parameters will need to be in agreement with the model
50  // being used.
51  // q_sensorrig_accelerometer:
52  // Rotation from sensorrig frame to accelerometer frame as a quaternion.
53  // t_sensorrig_accelerometer:
54  // Position of accelerometer relative to sensorrig origin resolved in the
55  // sensorrig frame.
56  // latency:
57  // Sensor latency in seconds.
58  // gravity:
59  // Gravity vector resolved in the world frame.
60  // control_points:
61  // Control points for the entire pose trajectory.
62  template <typename T>
63  bool operator()(T const* const* parameters, T* residual) {
64  // Parse intrinsics.
65  const T* intrinsics_ptr =
66  static_cast<const T*>(&(parameters[static_cast<int>(
67  AccelerometerParameterIndices::kIntrinsicsIndex)][0]));
68  const int parameter_size = accelerometer_model_->NumberOfParameters();
69  const Eigen::VectorX<T> intrinsics = Eigen::Map<const Eigen::VectorX<T>>(
70  intrinsics_ptr, parameter_size);
71  // Parse extrinsics.
72  const Eigen::Map<const Eigen::Quaternion<T>> q_sensorrig_accelerometer(
73  &(parameters[static_cast<int>(
74  AccelerometerParameterIndices::kExtrinsicsRotationIndex)][0]));
75  const Eigen::Map<const Eigen::Vector3<T>> t_sensorrig_accelerometer(
76  &(parameters[static_cast<int>(
77  AccelerometerParameterIndices::kExtrinsicsTranslationIndex)][0]));
78  // Parse latency.
79  const T latency =
80  parameters[static_cast<int>(
81  AccelerometerParameterIndices::kLatencyIndex)][0];
82  // Parse gravity.
83  const Eigen::Map<const Eigen::Vector3<T>> gravity(
84  &(parameters[static_cast<int>(
85  AccelerometerParameterIndices::kGravityIndex)][0]));
86  // Parse sensor rig spline resolved in the world frame.
87  const int& num_control_points =
88  trajectory_evaluation_params_.num_control_points;
89  Eigen::MatrixX<T> control_points(num_control_points, 6);
90  for (int i = 0; i < num_control_points; ++i) {
91  control_points.row(i) = Eigen::Map<const Eigen::Vector<T, 6>>(&(parameters[
92  static_cast<int>(
93  AccelerometerParameterIndices::kSensorRigPoseSplineControlPointsIndex
94  ) + i][0]));
95  }
96  const Eigen::MatrixX<T> basis_matrix =
97  trajectory_evaluation_params_.basis_matrix.template cast<T>();
98  const T knot0 = static_cast<T>(trajectory_evaluation_params_.knot0);
99  const T knot1 = static_cast<T>(trajectory_evaluation_params_.knot1);
100  const T stamp =
101  static_cast<T>(trajectory_evaluation_params_.stamp) - latency;
102  // Evaluate the pose, pose rate, and pose acceleration.
103  const Eigen::Vector<T, 6> pose_vector = BSpline<6, T>::Evaluate(
104  control_points, knot0, knot1, basis_matrix, stamp, /*derivative=*/0);
105  const Eigen::Vector<T, 6> pose_dot_vector = BSpline<6, T>::Evaluate(
106  control_points, knot0, knot1, basis_matrix, stamp, /*derivative=*/1);
107  const Eigen::Vector<T, 6> pose_ddot_vector = BSpline<6, T>::Evaluate(
108  control_points, knot0, knot1, basis_matrix, stamp, /*derivative=*/2);
109  // Compute the kinematics of the accelerometer.
110  const Eigen::Vector3<T> phi_sensorrig_world = -pose_vector.head(3);
111  const Eigen::Vector3<T> phi_dot_sensorrig_world = -pose_dot_vector.head(3);
112  const Eigen::Vector3<T> phi_ddot_sensorrig_world = -pose_ddot_vector.head(3);
113  const Eigen::Vector3<T> ddt_world_sensorrig = pose_ddot_vector.tail(3);
114  T q_sensorrig_world_array[4];
115  ceres::AngleAxisToQuaternion(phi_sensorrig_world.data(),
116  q_sensorrig_world_array);
117  const Eigen::Quaternion<T> q_sensorrig_world(
118  q_sensorrig_world_array[0], q_sensorrig_world_array[1],
119  q_sensorrig_world_array[2], q_sensorrig_world_array[3]);
120 
121  const Eigen::Matrix3<T> J_sensorrig_world =
122  ExpSO3Jacobian(phi_sensorrig_world);
123  const Eigen::Matrix3<T> J_dot_sensorrig_world =
124  ExpSO3JacobianDot(phi_sensorrig_world, phi_dot_sensorrig_world);
125  const Eigen::Vector3<T> omega_sensorrig_world =
126  J_sensorrig_world * phi_dot_sensorrig_world;
127  const Eigen::Vector3<T> alpha_sensorrig_world =
128  J_dot_sensorrig_world * phi_dot_sensorrig_world +
129  J_sensorrig_world * phi_ddot_sensorrig_world;
130  const Eigen::Matrix3<T> Alpha = -Skew(alpha_sensorrig_world);
131  const Eigen::Matrix3<T> Omega = -Skew(omega_sensorrig_world);
132  const Eigen::Vector3<T> ddt_accelerometer_world_accelerometer =
133  q_sensorrig_accelerometer.inverse() *
134  (q_sensorrig_world * (ddt_world_sensorrig - gravity) +
135  (Omega * Omega + Alpha) * t_sensorrig_accelerometer);
136  // Project the sensor angular velocity through the accelerometer model.
137  const absl::StatusOr<Eigen::Vector3<T>> projection =
138  accelerometer_model_->Project(
139  intrinsics, ddt_accelerometer_world_accelerometer);
140  if (projection.ok()) {
141  Eigen::Map<Eigen::Vector3<T>> error(residual);
142  const Eigen::Vector3<T> measurement = measurement_.template cast<T>();
143  error = (measurement - *projection) * static_cast<T>(information_);
144  return true;
145  }
146  return false;
147  }

149  private:
150  Eigen::Vector3d measurement_;
151  double information_;

153  TrajectoryEvaluationParams trajectory_evaluation_params_;
154 };
