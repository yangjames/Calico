27 // how the gyroscope model is initialized.
28 class GyroscopeCostFunctor {

31  explicit GyroscopeCostFunctor(

44  // Parameters to the cost function:
45  // intrinsics:
46  // All parameters in the intrinsics model as an Eigen column vector.
47  // Order of the parameters will need to be in agreement with the model
48  // being used.
49  // q_sensorrig_gyroscope:
50  // Rotation from sensorrig frame to gyroscope frame as a quaternion.
51  // t_sensorrig_gyroscope:
52  // Position of gyroscope relative to sensorrig origin resolved in the
53  // sensorrig frame.
54  // latency:
55  // Sensor latency in seconds.
56  // control_points:
57  // Control points for the entire pose trajectory.
58  template <typename T>
59  bool operator()(T const* const* parameters, T* residual) {
60  // Parse intrinsics.
61  const T* intrinsics_ptr =
62  static_cast<const T*>(&(parameters[static_cast<int>(
63  GyroscopeParameterIndices::kIntrinsicsIndex)][0]));
64  const int parameter_size = gyroscope_model_->NumberOfParameters();
65  const Eigen::VectorX<T> intrinsics = Eigen::Map<const Eigen::VectorX<T>>(
66  intrinsics_ptr, parameter_size);
67  // Parse extrinsics.
68  const Eigen::Map<const Eigen::Quaternion<T>> q_sensorrig_gyroscope(
69  &(parameters[static_cast<int>(
70  GyroscopeParameterIndices::kExtrinsicsRotationIndex)][0]));
71  const Eigen::Map<const Eigen::Vector3<T>> t_sensorrig_gyroscope(
72  &(parameters[static_cast<int>(
73  GyroscopeParameterIndices::kExtrinsicsTranslationIndex)][0]));
74  // Parse latency.
75  const T latency =
76  parameters[static_cast<int>(
77  GyroscopeParameterIndices::kLatencyIndex)][0];
78  // Parse sensor rig spline resolved in the world frame.
79  const int& num_control_points =
80  trajectory_evaluation_params_.num_control_points;
81  Eigen::MatrixX<T> control_points(num_control_points, 6);
82  for (int i = 0; i < num_control_points; ++i) {
83  control_points.row(i) = Eigen::Map<const Eigen::Vector<T, 6>>(
84  &(parameters[static_cast<int>(
85  GyroscopeParameterIndices::kSensorRigPoseSplineControlPointsIndex
86  ) + i][0]));
87  }
88  const Eigen::MatrixX<T> basis_matrix =
89  trajectory_evaluation_params_.basis_matrix.template cast<T>();
90  const T knot0 = static_cast<T>(trajectory_evaluation_params_.knot0);
91  const T knot1 = static_cast<T>(trajectory_evaluation_params_.knot1);
92  const T stamp =
93  static_cast<T>(trajectory_evaluation_params_.stamp) - latency;
94  // Evaluate the pose and pose rate.
95  const Eigen::Vector<T, 6> pose_vector = BSpline<6, T>::Evaluate(
96  control_points, knot0, knot1, basis_matrix, stamp, /*derivative=*/0);
97  const Eigen::Vector<T, 6> pose_dot_vector = BSpline<6, T>::Evaluate(
98  control_points, knot0, knot1, basis_matrix, stamp, /*derivative=*/1);
99  // Compute the angular velocity of the gyroscope.
100  // TODO(yangjames): Also evaluate acceleration for g-sensitivity
101  // calculations.
102  const Eigen::Vector3<T> phi_sensorrig_world = -pose_vector.head(3);
103  const Eigen::Vector3<T> phi_dot_sensorrig_world = -pose_dot_vector.head(3);
104  const Eigen::Matrix3<T> J = ExpSO3Jacobian(phi_sensorrig_world);
105  const Eigen::Vector3<T> omega_sensorrig_world = J * phi_dot_sensorrig_world;
106  const Eigen::Vector3<T> omega_gyroscope_world =
107  -(q_sensorrig_gyroscope.inverse() * omega_sensorrig_world);
108  // Project the sensor angular velocity through the gyroscope model.
109  const absl::StatusOr<Eigen::Vector3<T>> projection =
110  gyroscope_model_->Project(intrinsics, omega_gyroscope_world);
111  if (projection.ok()) {
112  Eigen::Map<Eigen::Vector3<T>> error(residual);
113  const Eigen::Vector3<T> measurement = measurement_.template cast<T>();
114  error = (measurement - *projection) * static_cast<T>(information_);
115  return true;
116  }
117  return false;
118  }

120  private:
121  Eigen::Vector3d measurement_;
122  double information_;
123  std::unique_ptr<GyroscopeModel> gyroscope_model_;
124  TrajectoryEvaluationParams trajectory_evaluation_params_;
125 };
