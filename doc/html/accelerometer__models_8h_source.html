126  static absl::StatusOr<Eigen::Vector3<T>> Project(
127  const Eigen::VectorX<T>& intrinsics,
128  const Eigen::Vector3<T>& ddt_sensor_world_sensor) {
129  if (intrinsics.size() != kNumberOfParameters) {
130  return absl::InvalidArgumentError(absl::StrCat(
131  "Invalid accelerometer intrinsics size. Expected ",

133  }
134  const T& scale = intrinsics(0);
135  const Eigen::Ref<const Eigen::Vector3<T>> bias =
136  intrinsics.segment(1, 3);
