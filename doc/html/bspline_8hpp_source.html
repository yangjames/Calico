39 template<int N, typename T>
40 Eigen::Vector<T, N> BSpline<N, T>::Evaluate(
41  const Eigen::Ref<const Eigen::MatrixX<T>>& control_points_set, T knot0,
42  T knot1, const Eigen::MatrixX<T>& basis_matrix, T stamp,
43  int derivative) {
44  // Compute u.
45  const T dt = knot1 - knot0;
46  const T dt_inv = static_cast<T>(1.0) / dt;
47  const T u = (stamp - knot0) * dt_inv;
48  // Derivative of u with respect to t raised to n'th derivative power per
49  // chain rule.
50  T dnu_dtn = static_cast<T>(1.0);
51  for (int j = 0; j < derivative; ++j) {
52  dnu_dtn *= dt_inv;
53  }
54  // Construct U vector.
55  Eigen::Matrix<T, 1, Eigen::Dynamic> derivative_coeffs(
56  control_points_set.rows());
57  derivative_coeffs.setOnes();
58  derivative_coeffs.head(derivative).setZero();
59  Eigen::Matrix<T, 1, Eigen::Dynamic> U(control_points_set.rows());
60  U.setOnes();
61  for (int i = derivative; i < derivative_coeffs.size(); ++i) {
62  T coeff = static_cast<T>(1.0);
63  for (int j = i - derivative; j < i; ++j) {
64  coeff *= (j + 1);
65  }
66  derivative_coeffs(i) = coeff;
67  U(i) = (i > derivative) ? (u * U(i-1)) : U(i);
68  }
69  U = U.array() * derivative_coeffs.array() * dnu_dtn;
70  // Evaluate spline.
71  return (U * basis_matrix * control_points_set).transpose();
72 }

75 absl::StatusOr<std::vector<Eigen::Vector<T, N>>> BSpline<N, T>::Interpolate(
76  const std::vector<T>& times, int derivative) const {
77  if (derivative < 0 || derivative > spline_degree_) {
78  return absl::InvalidArgumentError("Invalid derivative for interpolation.");
79  }
80  for (const T& t : times) {
81  if (t < valid_knots_.front() || t > valid_knots_.back()) {
82  return absl::InvalidArgumentError(absl::StrCat(
83  "Cannot interpolate ", t, ". Value is not within valid knots."));
84  }
85  }
86  int num_interp = times.size();
87  std::vector<Eigen::Vector<T, N>> y(num_interp);
88  for (int i = 0; i < num_interp; ++i) {
89  const int spline_idx = GetSplineIndex(times.at(i));
90  const int knot_idx = GetKnotIndexFromSplineIndex(spline_idx);
91  const Eigen::MatrixX<T> UM = GetSplineBasis(spline_idx, times.at(i), derivative);
92  Eigen::MatrixX<T> control_points(spline_order_, N);
93  for (int j = 0; j < spline_order_; ++j) {
94  control_points.row(j) = control_points_.at(spline_idx + j).transpose();
95  }
96  y[i] = Evaluate(control_points, knots_[knot_idx], knots_[knot_idx + 1],
97  Mi_[spline_idx], times[i], derivative);
98  }
99  return y;
100 }

104  int spline_idx, T stamp, int derivative) const {
105  // Compute u.
106  const int knot_idx = GetKnotIndexFromSplineIndex(spline_idx);
107  const T& knot0 = knots_.at(knot_idx);
108  const T& knot1 = knots_.at(knot_idx + 1);
109  const T dt = knot1 - knot0;
110  const T dt_inv = static_cast<T>(1.0) / dt;
111  const T u = (stamp - knot0) * dt_inv;
112  // Derivative of u with respect to t raised to n'th derivative power per
113  // chain rule.
114  T dnu_dtn = static_cast<T>(1.0);
115  for (int j = 0; j < derivative; ++j) {
116  dnu_dtn *= dt_inv;
117  }
118  // Construct U vector.

120  derivative_coeffs.setOnes();

122  Eigen::Matrix<T, 1, Eigen::Dynamic> U(spline_order_);
123  U.setOnes();

163 template <int N, typename T>
164 void BSpline<N, T>::ComputeKnotVector() {
165  const T duration = time_.back() - time_.front();
166  const T dt = 1.0 / knot_frequency_;
167  const int num_valid_knots = 1 + std::ceil(duration * knot_frequency_);
168  const int num_knots = num_valid_knots + 2 * spline_degree_;
169 
170  knots_.resize(num_knots);
171  valid_knots_.resize(num_valid_knots);
172  for (int i = -spline_degree_; i < num_knots - spline_degree_; ++i) {
173  const T knot_value = time_.front() + dt * i;
174  knots_[i + spline_degree_] = knot_value;
175  int valid_knot_index = i;
176  if (valid_knot_index > -1 && valid_knot_index < num_valid_knots) {
177  valid_knots_[valid_knot_index] = knot_value;
178  }
179  }

191 template<int N, typename T>
192 Eigen::MatrixX<T> BSpline<N, T>::M(int k, int i) {
193  Eigen::MatrixX<T> M_k;
194  if (k == 1) {
195  M_k.resize(k, k);
196  M_k(0, 0) = k;
197  return M_k;
198  }
199 
200  Eigen::MatrixX<T> M_km1 = M(k - 1, i);
201  const int num_rows = M_km1.rows();
202  const int num_cols = M_km1.cols();
203  Eigen::MatrixX<T> M1(num_rows + 1, num_cols), M2(num_rows + 1, num_cols);
204  M1.setZero();
205  M2.setZero();
206  M1.block(0, 0, num_rows, num_cols) = M_km1;
207  M2.block(1, 0, num_rows, num_cols) = M_km1;
208 
209  Eigen::MatrixX<T> A(k - 1,k);
210  Eigen::MatrixX<T> B(k - 1,k);
211  A.setZero();
212  B.setZero();
213  for (int index = 0; index < k - 1; ++index) {
214  int j = i - k + 2 + index;
215  const T d0 = d_0(k, i, j);
216  const T d1 = d_1(k, i, j);
217  A(index, index) = 1.0 - d0;
218  A(index, index + 1) = d0;
219  B(index, index) = -d1;
220  B(index, index + 1) = d1;
221  }
222  M_k = M1 * A + M2 * B;
223  return M_k;
224 }

246 template <int N, typename T>
247 void BSpline<N, T>::FitSpline() {
248  const int num_data = time_.size();
249  int num_control_points = knots_.size() - spline_order_;
250  Eigen::MatrixX<T> X(num_data, num_control_points);
251  X.setZero();
252  for (int j = 0; j < num_data; j++) {
253  const T t = time_[j];
254  int spline_index = -1;
255 
256  if (t == valid_knots_.back()) {
257  spline_index = Mi_.size() - 1;
258  }
259  else if (t == valid_knots_.front()) {
260  spline_index = 0;
261  }
262  else if (t < valid_knots_.back()) {
263  auto lower = std::upper_bound(valid_knots_.begin(), valid_knots_.end(), t);
264  spline_index = lower - valid_knots_.begin() - 1;
265  }
266  const int knot_index = GetKnotIndexFromSplineIndex(spline_index);
267  // Grab the intermediate knot values
268  const T ti = knots_[knot_index];
269  const T tii = knots_[knot_index + 1];
270  // Construct U vector
271  Eigen::VectorX<T> U(spline_order_);
272  U.setOnes();
273  const T u = (t - ti) / (tii - ti);
274  for (int i = 1; i < spline_order_; ++i) {
275  U(i) = u * U(i - 1);
276  }
277  // Construct U*M
278  Eigen::MatrixX<T> M = Mi_[spline_index];
279  Eigen::MatrixX<T> UM = U.transpose() * M;
280  X.block(j, spline_index, UM.rows(), UM.cols()) = UM;
281  }
282 
283  Eigen::Matrix<T,Eigen::Dynamic,N> data(num_data, N);
284  for (int i = 0; i < num_data; ++i) {
285  data.row(i) = data_[i].transpose();
286  }
287  // TODO(yangjames): X is highly sparse, and X'X is banded, symmetric, positive
288  // definite. Figure out how to sparsify the solve step for the control points
289  // as this gets very expensive with more knots.
290  const Eigen::MatrixX<T> XtX = X.transpose() * X;
291  const Eigen::MatrixX<T> Xtd = X.transpose() * data;
292  const Eigen::MatrixX<T> control_points =
293  XtX.colPivHouseholderQr().solve(Xtd);
294  for (int i = 0; i < control_points.rows(); ++i) {
295  control_points_.push_back(control_points.row(i).transpose());
296  }
297 }

299 template <int N, typename T>
300 absl::Status BSpline<N, T>::CheckDataForSplineFit(
301  const std::vector<T>& time,
302  const std::vector<Eigen::Vector<T, N>>& data,
303  int spline_order, T knot_frequency) {
304  // Assert that data and time are properly sized
305  if (!time.size()) {
306  return absl::InvalidArgumentError("Attempted to fit data on empty time vector.");
307  }
308  if (data.empty()) {
309  return absl::InvalidArgumentError("Attempted to fit on empty data.");
310  }
311  if (time.size() != data.size()) {
312  return absl::InvalidArgumentError("Data and time vectors are not the same size.");
313  }
314  // Check that time is strictly increasing
315  auto iter = std::adjacent_find(time_.begin(), time_.end(), std::greater<T>());
316  if (iter != time_.end()) {
317  return absl::InvalidArgumentError("Time vector is not monotonically increasing.");
318  }
319  // Check that spline order is greater than 2
320  if (spline_order < 2) {
321  return absl::InvalidArgumentError(absl::StrCat(
322  "Spline order must be greater than 2. Got ", spline_order));
323  }
324  // Check that knot frequency is greater than 0
325  if (knot_frequency <= 0) {
326  return absl::InvalidArgumentError("Knot frequency must be greater than 0.");
327  }
328  return absl::OkStatus();
329 }
