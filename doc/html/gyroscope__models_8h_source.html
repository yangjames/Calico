259  return absl::InvalidArgumentError(absl::StrCat(
260  "Invalid gyroscope intrinsics size. Expected ", kNumberOfParameters,
261  ", got ", intrinsics.size()));
262  }
263  const T& sx = intrinsics(0);
264  const T& sy = intrinsics(1);
265  const T& sz = intrinsics(2);
266  const T& a1 = intrinsics(3);
267  const T& a2 = intrinsics(4);
268  const T& a3 = intrinsics(5);
269  const T& a4 = intrinsics(6);
270  const T& a5 = intrinsics(7);
271  const T& a6 = intrinsics(8);
272  const T& fx = measurement.x();
273  const T& fy = measurement.y();
274  const T& fz = measurement.z();
275  const Eigen::Vector3<T> bias(intrinsics(9), intrinsics(10), intrinsics(11));
276  const Eigen::Vector3<T> d = bias - measurement;
277  const T det_A =
278  1.0 - a1*a3 - a2*a5 - a4*a6 + a1*a4*a5 + a2*a3*a6;
279  const T dx_inv_sx_detA = d.x() / (sx * det_A);
280  const T dy_inv_sy_detA = d.y() / (sy * det_A);
281  const T dz_inv_sz_detA = d.z() / (sz * det_A);
282  const T wx = (a4*a6 - 1) * dx_inv_sx_detA + (a1 - a2*a6) * dy_inv_sy_detA +
283  (a2 - a1*a4) * dz_inv_sz_detA;
284  const T wy = (a2*a5 - 1) * dy_inv_sy_detA + (a3 - a4*a5) * dx_inv_sx_detA +
285  (a4 - a2*a3) * dz_inv_sz_detA;
286  const T wz = (a1*a3 - 1) * dz_inv_sz_detA + (a5 - a3*a6) * dx_inv_sx_detA +
287  (a6 - a1*a5) * dy_inv_sy_detA;
288  return Eigen::Vector3<T>(wx, wy, wz);
289  }
