11 // Enum listing the positions of parameters for a camera cost function.
12 enum class CameraParameterIndices : int {
13  // Camera intrinsics.
14  kIntrinsicsIndex = 0,
15  // Extrinsic parameters of the camera relative to its sensor rig.
16  kExtrinsicsRotationIndex = 1,
17  kExtrinsicsTranslationIndex = 2,
18  // Sensor latency.
19  kLatencyIndex = 3,
20  // Parameters related to some detected "model" object in the world:
21  // 1. The point resolved in the model frame.
22  // 2. The rotation portion of the model pose resolved in the world
23  // frame.
24  // 3. The translation portion of the model pose resolved in the world
25  // frame.
26  kModelPointIndex = 4,
27  kModelRotationIndex = 5,
28  kModelTranslationIndex = 6,
29  // Rotation and position control points of the associated spline segment as an
30  // Nx6 matrix where N is the spline order.
31  kSensorRigPoseSplineControlPointsIndex = 7,

51  // Parameters to the cost function:
52  // intrinsics:
53  // All parameters in the intrinsics model as an Eigen column vector.
54  // Order of the parameters will need to be in agreement with the model
55  // being used.
56  // q_sensorrig_camera:
57  // Rotation from sensorrig frame to camera frame as a quaternion.
58  // t_sensorrig_camera:
59  // Position of camera relative to sensorrig origin resolved in the
60  // sensorrig frame.
61  // latency:
62  // Sensor latency in seconds.
63  // q_world_model:
64  // Rotation from world frame to model frame as a quaternion.
65  // t_model_point:
66  // Position of the point in the model resolved int he model frame.
67  // t_world_model:
68  // Position of model relative to world origin resolved in the world frame.
69  // control_points:
70  // Control points for the entire pose trajectory.
71  template <typename T>
72  bool operator()(T const* const* parameters, T* residual) {
73  // Parse intrinsics.
74  const T* intrinsics_ptr =
75  static_cast<const T*>(&(parameters[static_cast<int>(
76  CameraParameterIndices::kIntrinsicsIndex)][0]));
77  const int parameter_size = camera_model_->NumberOfParameters();
78  const Eigen::VectorX<T> intrinsics = Eigen::Map<const Eigen::VectorX<T>>(
79  intrinsics_ptr, parameter_size);
80  // Parse extrinsics.
81  const Eigen::Map<const Eigen::Quaternion<T>> q_sensorrig_camera(
82  &(parameters[static_cast<int>(
83  CameraParameterIndices::kExtrinsicsRotationIndex)][0]));
84  const Eigen::Map<const Eigen::Vector3<T>> t_sensorrig_camera(
85  &(parameters[static_cast<int>(
86  CameraParameterIndices::kExtrinsicsTranslationIndex)][0]));
87  // Parse latency.
88  const T latency =
89  parameters[static_cast<int>(CameraParameterIndices::kLatencyIndex)][0];
90  // Parse model point and model pose resolved in the world frame.
91  const Eigen::Map<const Eigen::Vector3<T>> t_model_point(
92  &(parameters[static_cast<int>(
93  CameraParameterIndices::kModelPointIndex)][0]));
94  const Eigen::Map<const Eigen::Quaternion<T>> q_world_model(
95  &(parameters[static_cast<int>(
96  CameraParameterIndices::kModelRotationIndex)][0]));
97  const Eigen::Map<const Eigen::Vector3<T>> t_world_model(
98  &(parameters[static_cast<int>(
99  CameraParameterIndices::kModelTranslationIndex)][0]));
100  // Parse sensor rig spline resolved in the world frame.
101  const int num_control_points =
102  trajectory_evaluation_params_.num_control_points;
103  Eigen::MatrixX<T> control_points(num_control_points, 6);
104  for (int i = 0; i < num_control_points; ++i) {
105  control_points.row(i) = Eigen::Map<const Eigen::Vector<T, 6>>(
106  &(parameters[static_cast<int>(
107  CameraParameterIndices::kSensorRigPoseSplineControlPointsIndex) + i
108  ][0]));
109  }
110  const Eigen::MatrixX<T> basis_matrix =
111  trajectory_evaluation_params_.basis_matrix.template cast<T>();
112  const T knot0 = static_cast<T>(trajectory_evaluation_params_.knot0);
113  const T knot1 = static_cast<T>(trajectory_evaluation_params_.knot1);
114  const T stamp =
115  static_cast<T>(trajectory_evaluation_params_.stamp) - latency;
116  // Evaluate the pose.
117  const Eigen::Vector<T, 6> pose_vector =
118  BSpline<6, T>::Evaluate(
119  control_points, knot0, knot1, basis_matrix, stamp, 0);
120  const Eigen::Vector3<T> phi_sensorrig_world = -pose_vector.head(3);
121  T q_sensorrig_world_array[4];
122  ceres::AngleAxisToQuaternion(phi_sensorrig_world.data(), q_sensorrig_world_array);
123  const Eigen::Quaternion<T> q_sensorrig_world(
124  q_sensorrig_world_array[0], q_sensorrig_world_array[1],
125  q_sensorrig_world_array[2], q_sensorrig_world_array[3]);
126  const Eigen::Vector3<T> t_world_sensorrig(pose_vector.data() + 3);
127  // Resolve the model point in the camera frame.
128  const Eigen::Quaternion<T> q_camera_model =
129  q_sensorrig_camera.inverse() * q_sensorrig_world * q_world_model;
130  const Eigen::Vector3<T> t_world_camera =
131  t_world_sensorrig + q_sensorrig_world.inverse() * t_sensorrig_camera;
132  const Eigen::Vector3<T> t_model_camera =
133  q_world_model.inverse() * (t_world_camera - t_world_model);
134  const Eigen::Vector3<T> t_camera_point =
135  q_camera_model * (t_model_point - t_model_camera);
136  // Project the point through the camera model.
137  const absl::StatusOr<Eigen::Vector2<T>> projection =
138  camera_model_->ProjectPoint(intrinsics, t_camera_point);
139  // Assign the residual, or return boolean indicating success/failure.
140  if (projection.ok()) {
141  Eigen::Map<Eigen::Vector2<T>> error(residual);
142  const Eigen::Vector2<T> pixel = pixel_.template cast<T>();
143  error = pixel - *projection;
144  return true;
145  }
146  return false;
147  }

149  private:

151  std::unique_ptr<CameraModel> camera_model_;
152  TrajectoryEvaluationParams trajectory_evaluation_params_;
